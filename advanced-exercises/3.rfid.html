RFID

                            

#include <ESP8266WiFi.h>
#include <Wire.h>
#include <LiquidCrystal_I2C.h>
#include <SPI.h>
#include <MFRC522.h>

LiquidCrystal_I2C lcd(0x27, 16, 2);

constexpr uint8_t RST_PIN = D3;     // Configurable, see typical pin layout above
constexpr uint8_t SS_PIN = D4;

byte readCard[4];
String tagID = "";
MFRC522 rfid(SS_PIN, RST_PIN);
MFRC522::MIFARE_Key key;

boolean getID();

void setup() {
pinMode(D8, OUTPUT);

    SPI.begin();
    rfid.PCD_Init();
    Wire.begin();
    lcd.backlight();
    lcd.setCursor(0, 0);
    lcd.print("    LCD and Rfid    ");
    lcd.setCursor(0, 1);
    lcd.print("  NodeMCU           ");
    delay(10000);
    lcd.clear();
    Serial.begin(9600);
    lcd.clear();
}

void loop() {

    lcd.setCursor(2, 0);
    lcd.print("Place your Card");

if (getID()) {
    lcd.clear();
    lcd.setCursor(0, 0);
    lcd.print("Reading Card");
    lcd.setCursor(-3, 1);
    lcd.print("Card No.: ");
    lcd.print(tagID);
    delay(2000);

    lcd.clear();
}
}

boolean getID() {

    if (! rfid.PICC_IsNewCardPresent()){
    return false;
    }
    if (!rfid.PICC_ReadCardSerial()) {
    return false;
    }
    tagID = "";
    for (uint8_t i = 0; i < 4; i++) {
        tagID.concat(String(rfid.uid.uidByte[i], HEX));
    }
    tagID.toUpperCase();
    rfid.PICC_HaltA();
    return true;

}
